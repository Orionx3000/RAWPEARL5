cmake_minimum_required(VERSION 3.22)
project(RAWPEARL5 VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# ─── Fetch JUCE 8 ─────────────────────────────────────────────────────────────
if (WIN32)
    set(JUCE_WEB_BROWSER ON CACHE BOOL "Enable Web Browser" FORCE)
    set(JUCE_USE_WIN_WEBVIEW2 ON CACHE BOOL "Enable WebView2" FORCE)
    set(JUCE_USE_WIN_WEBVIEW2_WITH_STATIC_LINKING ON CACHE BOOL "Enable static WebView2" FORCE)
endif()

include(FetchContent)
FetchContent_Declare(
    JUCE
    GIT_REPOSITORY https://github.com/juce-framework/JUCE.git
    GIT_TAG        8.0.7
    GIT_SHALLOW    TRUE
    GIT_PROGRESS   TRUE
)
FetchContent_MakeAvailable(JUCE)

# ─── RAWPEARL5 Application ────────────────────────────────────────────────────
juce_add_gui_app(RAWPEARL5
    PRODUCT_NAME        "RAWPEARL5"
    VERSION             "1.0.0"
    COMPANY_NAME        "Orion"
    BUNDLE_ID           "com.orion.rawpearl5"
    ICON                resources/RAWPEARL5.ico
    NEEDS_CURL          FALSE
    NEEDS_WEB_BROWSER   TRUE
    NEEDS_WEBVIEW2      TRUE
)

# ─── Source Files ─────────────────────────────────────────────────────────────
target_sources(RAWPEARL5 PRIVATE
    Source/Main.cpp
    Source/MainComponent.cpp
    Source/AudioEngine.cpp
    Source/BinaryData.cpp # Generated by gen_binary_data.py
)

target_include_directories(RAWPEARL5 PRIVATE Source)

# ─── JUCE Libraries ───────────────────────────────────────────────────────────
target_link_libraries(RAWPEARL5
    PRIVATE
        juce::juce_audio_basics
        juce::juce_audio_devices
        juce::juce_audio_formats
        juce::juce_audio_utils
        juce::juce_dsp
        juce::juce_gui_extra
        juce::juce_events
        juce::juce_graphics
        juce::juce_data_structures
        juce::juce_core
    PUBLIC
        juce::juce_recommended_config_flags
        juce::juce_recommended_lto_flags
        juce::juce_recommended_warning_flags
)

if (WIN32)
    target_compile_definitions(RAWPEARL5 PUBLIC JUCE_USE_WIN_WEBVIEW2=1)
endif()

# ─── Compile Definitions ─────────────────────────────────────────────────────
target_compile_definitions(RAWPEARL5
    PRIVATE
        JUCE_USE_CURL=0
        JUCE_WASAPI=1
        JUCE_DIRECTSOUND=1
        JUCE_WASAPI_EXCLUSIVE=0
        JUCE_ASIO=0              
        JUCE_DISPLAY_SPLASH_SCREEN=0
        JUCE_REPORT_APP_USAGE=0
        JUCE_APPLICATION_NAME_STRING="$<TARGET_PROPERTY:RAWPEARL5,JUCE_PRODUCT_NAME>"
        JUCE_APPLICATION_VERSION_STRING="$<TARGET_PROPERTY:RAWPEARL5,JUCE_VERSION>"
)

juce_generate_juce_header(RAWPEARL5)
